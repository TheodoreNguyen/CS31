(3)WHILE STATEMENT CONVERSION INNER LOOP

#include <iostream>
using namespace std;

int main()
{
	int len;

	cout << "Enter a number: ";
	cin >> len;

	for (int i = 0; i < len; i++)
	{
		int j = i + 1;
		while (j < len)
		{
			cout << " ";
			j++;
		}
		cout << "#" << endl;
	}
}




(4)DO-WHILE STATEMENT CONVERSION OUTER LOOP

#include <iostream>
using namespace std;

int main()
{
	int len;

	cout << "Enter a number: ";
	cin >> len;

	int i = 0;
	do
	{
		if (len > 0)
		{
			int j = i + 1;
			while (j < len)
			{
				cout << " ";
				j++;
			}
			cout << "#" << endl;
			i++;
		}
	} while (i < len);
}
